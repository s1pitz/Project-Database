CREATE TABLE TMember (
    IDMember char(4),
    NamaMember varchar(20),
    NoTelpon int,
    TanggalLahir date,
    PRIMARY KEY (IDMember)
);

CREATE TABLE TBuku (
    IDBuku char(4),
    JudulBuku varchar(20),
    Genre varchar(20),
    TahunTerbit int,
    PRIMARY KEY (IDBuku)
);

CREATE TABLE TPenerbit (
    IDPenerbit char(4),
    NamaPenerbit varchar(20),
    IDBuku char(4),
    PRIMARY KEY (IDPenerbit),
    FOREIGN KEY (IDBuku) REFERENCES TBuku(IDBuku)
);

CREATE TABLE TPenulis (
    IDPenulis char(4),
    NamaPenulis varchar(20),
    IDBuku char(4),
    PRIMARY KEY (IDPenulis),
    FOREIGN KEY (IDBuku) REFERENCES TBuku(IDBuku)
);

CREATE TABLE TMeminjam (
    IDMember char(4),
    IDBuku char(4),
    TanggalPeminjaman date,
    TanggalPengembalian date,
    FOREIGN KEY (IDMember) REFERENCES TMember(IDMember),
    FOREIGN KEY (IDBuku) REFERENCES TBuku(IDBuku)
);
